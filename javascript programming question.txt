function findDuplicates(array) {
  return array.filter((item, index) => array.indexOf(item) !== index);
}

let num = [12345]
function reverseNumber(num){
  var rev = 0;
  
  while(num > 0 ){
      let rem = num%10;
      rev = rev*10+rem;
      num = Math.floor(num/10)
  }
  return rev;
    
}
const revNumber = reverseNumber(num)
console.log(revNumber)
